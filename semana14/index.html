# --- FUNCIONES ---
def calcular_tasa(asistidas, fallidas):
    """Calcula tasa de retención en porcentaje."""
    total = asistidas + fallidas
    if total == 0:
        return 0.0
    return round((asistidas / total) * 100, 1)

def clasificar_paciente(tasa):
    """Clasifica estado del paciente según tasa."""
    if tasa >= 80:
        return "🟢 VERDE"
    elif tasa >= 50:
        return "🟡 AMARILLO"
    else:
        return "🔴 ROJO"

def generar_reporte(pacientes):
    """Genera resumen ejecutivo para la junta directiva."""
    print("
" + "="*50)
    print("  REPORTE EJECUTIVO — NEUROBIZ S.A.S.")
    print("="*50)

    conteo = {"🟢 VERDE": 0, "🟡 AMARILLO": 0, "🔴 ROJO": 0}
    suma_tasas = 0

    for p in pacientes:
        tasa = calcular_tasa(p['asistidas'], p['fallidas'])
        estado = clasificar_paciente(tasa)
        conteo[estado] = conteo.get(estado, 0) + 1
        suma_tasas += tasa
        print(f"  {p['nombre']:20} {tasa:5.1f}%  {estado}")

    print("-"*50)
    promedio = suma_tasas / len(pacientes) if pacientes else 0
    print(f"  Pacientes analizados: {len(pacientes)}")
    print(f"  Tasa promedio:        {promedio:.1f}%")
    print(f"  Verde:  {conteo['🟢 VERDE']} | Amarillo: {conteo['🟡 AMARILLO']} | Rojo: {conteo['🔴 ROJO']}")
    print("="*50)

# --- PROGRAMA PRINCIPAL ---
